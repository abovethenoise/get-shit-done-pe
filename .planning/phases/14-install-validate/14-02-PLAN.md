**1. Add post-install token validation (INST-02 safety net):**
After all file copy operations complete, add a validation scan:
```javascript
function validateNoUnresolvedTokens(dir) {
  const failures = [];
  function scan(d) {
    for (const entry of fs.readdirSync(d, { withFileTypes: true })) {
      const full = path.join(d, entry.name);
      if (entry.isDirectory()) scan(full);
      else if (entry.name.endsWith('.md') || entry.name.endsWith('.js') || entry.name.endsWith('.json')) {
        const content = fs.readFileSync(full, 'utf8');
        if (content.includes('{GSD_ROOT}')) failures.push(full);
      }
    }
  }
  scan(dir);
  return failures;
}
```
